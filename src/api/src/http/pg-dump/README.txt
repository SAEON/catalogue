** Instructions **

**** RESTORE DATABASE ****
Create a new database, and make sure PostGIS extensions are installed:

create extension if not exists address_standardizer;
create extension if not exists address_standardizer_data_us;
create extension if not exists bloom;
create extension if not exists btree_gin;
create extension if not exists btree_gist;
create extension if not exists cube;
create extension if not exists fuzzystrmatch;
create extension if not exists intarray;
create extension if not exists pg_trgm;
create extension if not exists pgcrypto;
create extension if not exists postgis;
create extension if not exists postgis_raster;
create extension if not exists postgis_sfcgal;
create extension if not exists postgis_tiger_geocoder;
create extension if not exists postgis_topology;

**** Load raster files into PostGIS ****
All raster files are included in this zip archive. They need to each be restored to PostGIS using the raster2pgsql command (run this command for each raster file). 

(NOTE - it's helpful to rename the schema that is generated by the backup process to something useful)

raster2pgsql -d -F -q -R -I -f raster_data <absolute path to raster> <your schema>.<table name> -c | psql --dbname <db name> --host <host> --port <port> --username <your username>
